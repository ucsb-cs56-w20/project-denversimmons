package edu.ucsb.cs56.w20.lab07.formbeans;

public class EqSearch
{

	minmag = 0;
    }

    {
	distance = d;
    }

    public void setMinmag(int m)
    {
	minmag = m;

	return distance;
    }

    public int getMinmag()
    {
	return minmag;
    }
}
